‚öô Epoch Distributor

Run this script to build the Merkle tree and submit the epoch root on-chain. Generates proofs for all workers.

ethers v6
merkletreejs
scripts/epoch-distributor.js
Copy
const { ethers } = require('ethers');
const MerkleTree  = require('merkletreejs').MerkleTree;
const keccak256   = require('keccak256');
const fs          = require('fs');
require('dotenv').config();

const RPC   = 'https://opbnb-testnet-rpc.bnbchain.org';
const VAULT = process.env.NEXT_PUBLIC_VAULT;
const ABI   = ['function submitEpochRoot(bytes32) external'];

async function distributeEpoch() {
  const provider = new ethers.JsonRpcProvider(RPC);
  const wallet   = new ethers.Wallet(process.env.PRIVATE_KEY, provider);
  const vault    = new ethers.Contract(VAULT, ABI, wallet);

  // ‚îÄ‚îÄ Worker rewards from your DB ‚îÄ‚îÄ
  const rewards = [
    { address: '0xYourWallet1', usdc: ethers.parseUnits('5.00', 6) },
    { address: '0xYourWallet2', usdc: ethers.parseUnits('3.20', 6) },
  ];

  // ‚îÄ‚îÄ Build Merkle Tree ‚îÄ‚îÄ
  const leaves = rewards.map(({ address, usdc }) =>
    keccak256(Buffer.from(
      ethers.solidityPackedKeccak256(
        ['address', 'uint256'], [address, usdc]
      ).slice(2), 'hex'
    ))
  );

  const tree = new MerkleTree(leaves, keccak256, { sortPairs: true });
  const root = tree.getHexRoot();
  console.log('üå≥ Merkle root:', root);

  // ‚îÄ‚îÄ Submit on-chain ‚îÄ‚îÄ
  const tx = await vault.submitEpochRoot(root);
  await tx.wait();
  console.log('‚úÖ Epoch TX:', tx.hash);

  // ‚îÄ‚îÄ Save proofs ‚îÄ‚îÄ
  const proofs = {};
  rewards.forEach(({ address, usdc }, i) => {
    proofs[address] = { amount: usdc.toString(), proof: tree.getHexProof(leaves[i]) };
  });
  fs.writeFileSync('./epoch-proofs.json', JSON.stringify({ root, proofs }, null, 2));
  console.log('üìù Saved epoch-proofs.json');
}

distributeEpoch().catch(console.error);
